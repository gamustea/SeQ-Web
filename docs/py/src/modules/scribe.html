src.modules.scribe

src.modules.scribe — Generación de contenido con IA.

Módulo transversal que unifica todo el "digest" de la plataforma: cualquier módulo (Aegis, Sentinel, …) construye un AIInput y se lo pasa a un AIGenerator, que delega en una estrategia de model calling inyectada (Ollama en local o OpenAI en la nube). La estrategia se decide por módulo desde SecOpsConfig.json mediante build_generator.

Uso típico:

from src.modules.scribe import build_generator, AIInput generator = build_generator("aegis") result = generator.digest(AIInput(system_prompt=..., user_prompt=...)) data = result.parse_json()

53        repeat_penalty: Penalización por repetición (Ollama). Ignorado por OpenAI.

66    def to_messages(self) -> list[dict]:
67        """Construye la lista de mensajes en formato chat (rol/contenido)."""
68        messages: list[dict] = [{"role": "system", "content": self.system_prompt}]
69        for ex in self.examples:

 88    def parse_json(self, *, detect_truncation: bool = True) -> dict:
 89        """
 90        Parseo JSON robusto con varias estrategias de recuperación.
 91
 92        Centraliza la lógica que antes duplicaban los writers de dominio:
 93            1. Detección temprana de truncado (la llave raíz no cierra).
 94            2. Intento directo.
 95            3. Extracción por patrones (bloques markdown, ``{...}``).
 96            4. Limpieza de texto sobrante antes/después del objeto.
 97
 98        Args:
 99            detect_truncation: Si True, lanza error cuando el JSON no cierra
100                su llave raíz (respuesta cortada por límite de tokens).
101
102        Returns:
103            El diccionario parseado.
104
105        Raises:
106            AIResponseError: Si la respuesta es vacía, truncada o no parseable.
107        """
108        raw = self.text
109        if not raw:
110            raise AIResponseError("Respuesta vacía del modelo")
111
112        stripped = raw.rstrip()
113        if detect_truncation and stripped and stripped[-1] != "}":
114            logger.warning(
115                "Respuesta truncada detectada (num_predict alcanzado). "
116                "Últimos 80 chars: %r",
117                stripped[-80:],
118            )
119            raise AIResponseError(
120                "Respuesta truncada por límite de tokens: el JSON no está completo. "
121                "Aumenta num_predict o reduce el tamaño del prompt."
122            )
123
124        try:
125            return json.loads(raw)
126        except json.JSONDecodeError:
127            pass
128
129        for pattern in (
130            r"```(?:json)?\s*([\s\S]*?)\s*```",
131            r"JSON:\s*(\{[\s\S]*\})",
132            r"\{[\s\S]*\}",
133        ):
134            match = re.search(pattern, raw)
135            if match:
136                try:
137                    return json.loads(match.group(1) if match.groups() else match.group())
138                except json.JSONDecodeError:
139                    continue
140
141        cleaned = re.sub(r"^[^{]*", "", raw)
142        cleaned = re.sub(r"[^}]*$", "", cleaned)
143        try:
144            return json.loads(cleaned)
145        except json.JSONDecodeError as exc:
146            logger.error("No se pudo parsear respuesta: %s", raw[:500])
147            raise AIResponseError(f"JSON inválido tras limpieza: {exc}")

class OllamaStrategy(src.modules.scribe.ModelStrategy): 
 53class OllamaStrategy(ModelStrategy):
 54    """Estrategia que llama a un modelo local servido por Ollama."""
 55
 56    name = "ollama"
 57
 58    def __init__(self, host: str, model: str, timeout: int = 300) -> None:
 59        import ollama
 60
 61        self.host = host
 62        self.model = model
 63        logger.info("[scribe/ollama] cliente host=%s model=%s", host, model)
 64        self._client = ollama.Client(host=host, timeout=timeout)
 65
 66    def _options(self, ai_input: AIInput) -> dict:
 67        return {
 68            "num_predict": ai_input.num_predict,
 69            "temperature": ai_input.temperature,
 70            "top_p": ai_input.top_p,
 71            "repeat_penalty": ai_input.repeat_penalty,
 72        }
 73
 74    def complete(self, ai_input: AIInput, tool_executor: Optional[ToolExecutor] = None) -> str:
 75        messages = ai_input.to_messages()
 76        options = self._options(ai_input)
 77        fmt = "json" if ai_input.json_mode else None
 78
 79        try:
 80            resp = self._client.chat(
 81                model=self.model,
 82                messages=messages,
 83                tools=ai_input.tools,
 84                format=fmt,
 85                options=options,
 86            )
 87
 88            tool_calls = getattr(resp.message, "tool_calls", None)
 89            if tool_calls and tool_executor:
 90                logger.info("[scribe/ollama] tool_calls: %d", len(tool_calls))
 91                messages.append({
 92                    "role": "assistant",
 93                    "content": resp.message.content or "",
 94                    "tool_calls": tool_calls,
 95                })
 96                for tc in tool_calls:
 97                    args = tc.function.arguments or {}
 98                    result = tool_executor(tc.function.name, dict(args))
 99                    messages.append({"role": "tool", "content": result})
100
101                resp = self._client.chat(
102                    model=self.model,
103                    messages=messages,
104                    format=fmt,
105                    options=options,
106                )
107
108            return (resp.message.content or "").strip()
109
110        except Exception as exc:
111            logger.error("[scribe/ollama] error en %s: %s", self.host, exc, exc_info=True)
112            raise AIConnectionError(str(exc), model=self.model) from exc

115class OpenAIStrategy(ModelStrategy):
116    """Estrategia que llama a la API de OpenAI (p.ej. gpt-4o-mini)."""
117
118    name = "openai"
119
120    def __init__(
121        self,
122        api_key: str,
123        model: str,
124        base_url: Optional[str] = None,
125        timeout: int = 120,
126    ) -> None:
127        from openai import OpenAI
128
129        self.model = model
130        logger.info("[scribe/openai] cliente model=%s base_url=%s", model, base_url or "default")
131        self._client = OpenAI(api_key=api_key, base_url=base_url or None, timeout=timeout)
132
133    def _create(self, messages: list[dict], ai_input: AIInput, with_tools: bool):
134        kwargs: dict[str, Any] = {
135            "model": self.model,
136            "messages": messages,
137            "temperature": ai_input.temperature,
138            "max_tokens": ai_input.num_predict,
139            "top_p": ai_input.top_p,
140        }
141        if ai_input.json_mode:
142            kwargs["response_format"] = {"type": "json_object"}
143        if with_tools and ai_input.tools:
144            kwargs["tools"] = ai_input.tools
145        return self._client.chat.completions.create(**kwargs)
146
147    def complete(self, ai_input: AIInput, tool_executor: Optional[ToolExecutor] = None) -> str:
148        messages = ai_input.to_messages()
149
150        try:
151            resp = self._create(messages, ai_input, with_tools=True)
152            message = resp.choices[0].message
153
154            tool_calls = getattr(message, "tool_calls", None)
155            if tool_calls and tool_executor:
156                logger.info("[scribe/openai] tool_calls: %d", len(tool_calls))
157                messages.append({
158                    "role": "assistant",
159                    "content": message.content or "",
160                    "tool_calls": [
161                        {
162                            "id": tc.id,
163                            "type": "function",
164                            "function": {
165                                "name": tc.function.name,
166                                "arguments": tc.function.arguments,
167                            },
168                        }
169                        for tc in tool_calls
170                    ],
171                })
172                for tc in tool_calls:
173                    try:
174                        args = json.loads(tc.function.arguments or "{}")
175                    except json.JSONDecodeError:
176                        args = {}
177                    result = tool_executor(tc.function.name, args)
178                    messages.append({
179                        "role": "tool",
180                        "tool_call_id": tc.id,
181                        "content": result,
182                    })
183
184                resp = self._create(messages, ai_input, with_tools=False)
185                message = resp.choices[0].message
186
187            return (message.content or "").strip()
188
189        except Exception as exc:
190            logger.error("[scribe/openai] error: %s", exc, exc_info=True)
191            raise AIConnectionError(str(exc), model=self.model) from exc

 74            if self._failures >= self._breaker_threshold:
 75                elapsed = time.time() - (self._last_failure_time or 0)
 76                if elapsed < self._breaker_timeout:

 78                self._failures = 0

 85        with self._lock:
 86            self._failures += 1
