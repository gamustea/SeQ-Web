357        """
358        try:

363            self.logger.warning(f"Error durante rollback: {e}")

366                    self.session.close()

370                        self.logger.info("Sesión recreada después de error en rollback")

185        self,
186        messages: list[dict],
187        tools: list[dict] | None = None,
188        temperature: float = 0.5,
189        num_predict: int = 4096,
190    ) -> ChatResponse:
191        """
192        Call the Ollama model with message history and optional tools.
193
194        Sends messages to the model and handles tool call responses
195        automatically by performing web searches when the model requests them.
196        Configures model parameters like temperature and max tokens.
197
198        Args:
199            messages: List of message dictionaries with 'role' and 'content'.
200            tools: Optional list of tool definitions for function calling.
201            temperature: Sampling temperature (default: 0.5).
202            num_predict: Maximum tokens to predict (default: 4096).
203
204        Returns:
205            ChatResponse object from Ollama with the model's reply.
206
207        Raises:
208            Exception: If connection to Ollama fails.
209        """
210        options = {
211            "num_predict":    num_predict,
212            "temperature":    temperature,
213            "top_p":          0.9,
214            "repeat_penalty": 1.1,
215        }
216        
217        self.logger.info(f"[Ollama] Calling model={self.model} at {self.host}")
218        
219        try:
220            self.logger.info(f"[Ollama] Sending request to {self.host}/api/chat")
221            resp = self._client.chat(
222                model    = self.model,
223                messages = messages,
224                tools    = tools,
225                format   = "json",
226                options  = options,
227            )
228            self.logger.info(f"[Ollama] Response received successfully")
229        except Exception as e:
230            self.logger.error(f"[Ollama] Error connecting to {self.host}: {e}")
231            raise
232        
233        if getattr(resp.message, "tool_calls", None):
234            messages.append({
235                "role":       "assistant",
236                "content":    resp.message.content or "",
237                "tool_calls": resp.message.tool_calls,
238            })
239            for tc in resp.message.tool_calls:
240                query         = tc.function.arguments.get("query", "")
241                search_result = self._web_search(query)
242                messages.append({"role": "tool", "content": search_result})
243            
244            resp = self._client.chat(
245                model    = self.model,
246                messages = messages,
247                format   = "json",
248                options  = options,
249            )
250        
251        return resp

Abstract base class for AI-powered content writers using Ollama.

Provides common infrastructure for all AI writers including:

  • Ollama client initialization and configuration
  • Web search integration with DuckDuckGo
  • Tool call handling and response processing
  • Error handling and retry logic

Subclasses must implement: - _build_system_prompt(): Domain-specific system prompt - _build_user_prompt(): User prompt with domain parameters - generate(): Main generation method returning domain content

If host or model are not provided, they are read from environment variables OLLAMA_HOST and OLLAMA_MODEL (or defaults).

Attributes: host: Ollama server URL. model: Ollama model name. logger: Logger instance for the class. _client: Ollama client instance.

Example:

...     def _build_system_prompt(self) -> str:

...     def _build_user_prompt(self, scan) -> str:
...         return f"Analyze this scan: {scan.target}"
...     def generate(self, scan):
...         prompt = self._build_user_prompt(scan)
...         messages = [
...             {"role": "system", "content": self._build_system_prompt()},
...             {"role": "user", "content": prompt}
...         ]
...         return self._call_model(messages)
