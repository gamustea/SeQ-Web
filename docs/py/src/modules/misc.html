152    def _resolve_config_path(cls) -> Path:
153        if cls._configs_path is not None:
154            return cls._configs_path
155        
156        this_file = Path(__file__).resolve()
157        
158        candidates = [
159            this_file.parent.parent.parent.parent / "SecOpsConfig.json",
160            this_file.parent.parent.parent.parent / "SecConfig.json",
161            this_file.parent.parent.parent / "SecOpsConfig.json",
162            this_file.parent.parent.parent / "SecConfig.json",
163            this_file.parent.parent / "SecOpsConfig.json",
164            this_file.parent.parent / "SecConfig.json",
165            this_file.parent / "SecOpsConfig.json",
166            this_file.parent / "SecConfig.json",
167        ]
168        
169        for candidate in candidates:
170            if candidate.exists():
171                cls._configs_path = candidate
172                return cls._configs_path

174        raise FileNotFoundError("No se encontró ningún archivo de configuración.")

246    @staticmethod
247    def get_directory_of(directory_type: DirectoryType) -> str:
248        """
249        Los directorios no son secretos → siempre desde el JSON,
250        pero permite sobrescribir via variables de entorno.
251        
252        Variables de entorno reconocidas:
253        - TEMP_DIR: directorio temporal
254        - LOG_DIR: directorio de logs
255        - OUTPUT_DIR: directorio de salida (sentinel/aegis)
256        - RESOURCE_DIR: directorio de recursos
257        
258        Args:
259            directory_type: DirectoryType enum member.
260            
261        Returns:
262            Absolute path string to the directory.
263        """
264        dir_key = directory_type.value
265        
266        env_mapping = {
267            "tempdir": "TEMP_DIR",
268            "logdir": "LOG_DIR",
269            "output": "OUTPUT_DIR",
270            "stack": "OUTPUT_DIR",
271            "resourcedir": "RESOURCE_DIR",
272        }
273        
274        env_var = env_mapping.get(dir_key)
275        if env_var:
276            env_value = os.getenv(env_var)
277            if env_value:
278                return env_value
279        
280        configs = ConfigReader._load_configs()
281        
282        if "." in dir_key:
283            parts = dir_key.split(".")
284            module_key = parts[0]
285            if module_key not in configs:
286                raise ValueError(f"Módulo '{module_key}' no encontrado en la configuración.")
287            module_config = configs[module_key]
288            if "directories" not in module_config:
289                raise ValueError(f"Directorio '{dir_key}' no encontrado en la configuración.")
290            directories = module_config["directories"]
291            sub_key = parts[1]
292            if sub_key not in directories:
293                raise ValueError(f"Directorio '{sub_key}' no encontrado en la configuración.")
294            raw_path = directories[sub_key]
295        else:
296            if dir_key not in configs.get("general", {}).get("directories", {}):
297                raise ValueError(f"Directorio '{dir_key}' no encontrado en la configuración.")
298            raw_path = configs["general"]["directories"][dir_key]
299
300        path = Path(raw_path)
301
302        if not path.is_absolute():
303            app_root = Path(__file__).resolve().parent.parent.parent
304            path = app_root / path
305
306        return str(path)

 74def normalize_target(
 75    user_input: str,
 76    resolve_hostname: bool = False,
 77    dns_timeout: float = _DNS_TIMEOUT,
 78) -> Tuple[Optional[str], Optional[str]]:
 79    """
 80    Normaliza el target del usuario a IP + hostname.
 81    Acepta IPs, dominios o URLs completas (http://, https://).
 82
 83    Args:
 84        user_input:        IP, dominio o URL completa.
 85        resolve_hostname:  Si es True y el input es una IP, intenta resolver
 86                           el hostname vía reverse DNS (con timeout acotado).
 87                           Si es False, el hostname se omite (se devuelve la IP
 88                           también en esa posición). Por defecto False.
 89        dns_timeout:       Segundos máximos para la resolución DNS inversa.
 90
 91    Returns:
 92        (ip, hostname): hostname == ip cuando no se resuelve o resolve_hostname=False.
 93    """
 94    cleaned_input = user_input.strip()
 95
 96    # Extraer host de una URL completa
 97    if "://" in cleaned_input:
 98        parsed = urlparse(cleaned_input)
 99        if not parsed.netloc and parsed.path:
100            cleaned_input = parsed.path.split('/')[0]
101        else:
102            cleaned_input = parsed.netloc.split(':')[0]
103    else:
104        cleaned_input = cleaned_input.split(':')[0].split('/')[0]
105
106    ip: Optional[str] = None
107    hostname: Optional[str] = None
108
109    try:
110        ip_obj = ipaddress.ip_address(cleaned_input)
111        ip = str(ip_obj)
112
113        if resolve_hostname:
114            # Reverse lookup acotado en tiempo; fallback a la propia IP
115            hostname = _gethostbyaddr_with_timeout(ip, dns_timeout) or ip
116        else:
117            hostname = ip   # No se necesita hostname: evitar la llamada bloqueante
118
119    except ValueError:
120        hostname = cleaned_input
121        try:
122            ip = socket.gethostbyname(hostname)  # Forward lookup: rápido y necesario
123        except socket.gaierror as e:
124            raise ValueError(f"No se pudo resolver '{user_input}': {e}")
125
126    return ip, hostname
