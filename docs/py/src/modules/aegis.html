 51    def generate(self, topic_id: int, tweaks: dict | None = None) -> int:
 52        """
 53        Lanza la generación asíncrona de una píldora y devuelve el documentId
 54        inmediatamente. Thread-safe.
 55        """
 56        with self._lock:
 57            tweaks      = tweaks or {}
 58            document_id = self._create_pending_document(topic_id)
 59
 60            thread_manager = self.__class__(self.user)
 61            threading.Thread(
 62                target  = thread_manager._run_generation_workflow,
 63                args    = (document_id, topic_id, tweaks),
 64                daemon  = True,
 65                name    = f"AegisGen-{document_id}",
 66            ).start()
 67
 68        return document_id

 72        if not doc:

 75        result = {

 77            "internalName": doc.title,
 78            "title": doc.subtitle or "Sin título",

638
