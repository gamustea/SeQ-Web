159class UnitOfWork:
160    """
161    Manages the lifecycle of a single SQLAlchemy session.
162
163    Wraps one session and exposes commit / rollback / close operations
164    with consistent error handling. Supports use as a context manager,
165    committing on clean exit and rolling back on exception.
166
167    Attributes:
168        session:        The underlying SQLAlchemy session.
169        _owns_session:  True if this UoW created the session (and must close it).
170
171    Example:
172    >>> with UnitOfWork() as uow:
173    ...     scan_repo = ScanRepository(uow)
174    ...     scan_repo.save(NmapScan(target="10.0.0.1", user_id=1))
175    """
176
177    def __init__(self, session: Optional[Session] = None) -> None:
178        """
179        Initialize the Unit of Work with an optional existing session.
180
181        Args:
182            session: Optional existing SQLAlchemy session. If not provided,
183                        a new session is obtained from the database module.
184        """
185        if session is not None:
186            self.session = session
187            self._owns_session = False
188        else:
189            self.session = get_session()
190            self._owns_session = True

196    def __enter__(self) -> UnitOfWork:
197        return self

199    def __exit__(self, exc_type, exc_val, exc_tb) -> bool:
200        """
201        Commit on clean exit, rollback on exception, always close.
202
203        Returns:
204            False — exceptions are never suppressed.
205        """
206        if exc_type is None:
207            self.commit()
208        else:
209            self.rollback()
210        self.close()
211        return False

217    def commit(self) -> None:
218        """
219        Commit the current transaction.
220
221        Raises:
222            SQLAlchemyError: If the commit fails. A rollback is performed
223                             automatically before re-raising.
224        """
225        try:
226            self.session.commit()
227        except SQLAlchemyError as e:
228            self.rollback()
229            raise SQLAlchemyError(f"Commit failed: {e}") from e
230
231    def rollback(self) -> None:
232        """
233        Roll back the current transaction.
234
235        If the rollback itself fails, the session is closed and recreated
236        so that subsequent operations on this UoW can still proceed.
237        """
238        try:
239            if self.session is not None:
240                self.session.rollback()
241        except Exception as rollback_err:
242            if self._owns_session:
243                try:
244                    self.session.close()
245                    self.session = get_session()
246                except Exception:
247                    pass
248            raise RuntimeError(
249                f"Rollback failed, session has been recreated: {rollback_err}"
250            ) from rollback_err
251
252    def close(self) -> None:
253        """
254        Close the session if this UoW owns it.
255
256        expunge_all() is called before closing so that ORM objects returned
257        by queries remain accessible after the UoW exits. Their already-loaded
258        attributes (including eagerly loaded relationships) stay intact, but
259        any subsequent lazy load attempt will raise DetachedInstanceError —
260        which is the correct and expected behaviour outside a session scope.
261
262        Safe to call multiple times; subsequent calls are no-ops.
263        """
264        if self._owns_session and self.session is not None:
265            try:
266                self.session.expunge_all()
267                self.session.close()
268                close_all()
269            finally:
270                self.session = None

 33    """
 34    Generic repository providing type-safe CRUD operations.
 35
 36    Concrete repositories inherit from this class, specifying the model
 37    type via the Generic parameter. All database access goes through the
 38    UnitOfWork's session, keeping transaction control in the caller.
 39
 40    Attributes:
 41        _model:  The SQLAlchemy model class this repository manages.
 42        _uow:    The Unit of Work providing the active session.
 43
 44    Type Parameters:
 45        T: SQLAlchemy declarative model class.
 46
 47    Example:
 48    >>> class UserRepository(BaseRepository[User]):
 49    ...     def __init__(self, uow: UnitOfWork) -> None:
 50    ...         super().__init__(User, uow)
 51    ...
 52    >>> with UnitOfWork() as uow:
 53    ...     repo = UserRepository(uow)
 54    ...     user = repo.get_by_id(42)
 55    """

 73    def _session(self):
 74        """Return the current session from the Unit of Work."""
 75        if self._uow.session is None:
 76            raise RuntimeError(
 77                f"{self.__class__.__name__}: No active session in UnitOfWork."
 78            )
 79        return self._uow.session

 87        Retrieve a single record by primary key.
 88

 97    def get_by_field(self, field: str, value: Any) -> Optional[T]:
 98        """
 99        Retrieve the first record matching a field/value pair.
100
101        Args:
102            field:  Attribute name on the model.
103            value:  Value to match.
104

107
108        Raises:
