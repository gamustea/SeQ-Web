 60    def analyze(self, raw_headers: str, user_id: int, title: str | None = None) -> int:
 61        """Submit raw email headers for background analysis.
 62
 63        Creates an IrisAnalysis record in ``pending`` state and enqueues
 64        a SeQueue task (category ``"iris.analyze"``) that runs every
 65        registered rule, aggregates scores, and persists the results.
 66
 67        Args:
 68            raw_headers: Full email header block as plain text.
 69            user_id:     Primary key of the requesting user.
 70            title:       Optional user-defined label for quick
 71                         identification in history.
 72
 73        Returns:
 74            The new IrisAnalysis primary key (``analysis_id``).  The
 75            caller should store this to later poll status or fetch the
 76            full report.
 77
 78        Raises:
 79            IrisInvalidInputError: If the headers contain fewer than
 80                the minimum required header lines (configurable via
 81                ``iris.min_headers`` in SecOpsConfig.json).
 82        """
 83        self._validate_headers_pre(raw_headers)
 84        analysis_id = self._create_analysis_record(raw_headers, user_id, title=title)
 85        self.logger.info(f"Iris analysis {analysis_id} created for user {user_id}")
 86
 87        cancel_event = threading.Event()
 88        thread_mgr = IrisManager()
 89
 90        SeQueue.get_instance().submit(
 91            func=thread_mgr._run_analysis,
 92            args=(analysis_id, raw_headers),
 93            name=f"IrisAnalysis-{analysis_id}",
 94            category="iris.analyze",
 95            external_id=f"iris-analysis:{analysis_id}",
 96            on_cancel=cancel_event.set,
 97        )
 98
 99        return analysis_id

209    def cancel_analysis(self, analysis_id: int, user_id: int) -> bool:
210        """Cancel a running or pending analysis.
211
212        Signals the SeQueue task to stop and marks the database record
213        as ``cancelled``.
214
215        Args:
216            analysis_id: Primary key of the analysis to cancel.
217            user_id:     Owner ID (must match the record's owner).
218
219        Returns:
220            True if the cancellation was successful.
221
222        Raises:
223            IrisAnalysisNotFoundError: If the analysis does not exist
224                or does not belong to this user.
225            IrisInvalidStateError: If the analysis is not in a
226                cancellable state (``pending`` or ``running``).
227        """
228        analysis = self.assert_analysis_ownership(analysis_id, user_id)
229
230        if analysis.status not in _CANCELLABLE_STATES:
231            raise IrisInvalidStateError(
232                f"Analysis {analysis_id} cannot be cancelled in state: {analysis.status}"
233            )
234
235        sequeue = SeQueue.get_instance()
236        sq_task = sequeue.get_task_by_external_id(
237            f"iris-analysis:{analysis_id}", category="iris.analyze"
238        )
239        if not sq_task:
240            self.logger.warning(f"No active task found for analysis {analysis_id}")
241            return False
242
243        cancelled = sequeue.cancel(sq_task.id)
244        if cancelled:
245            with UnitOfWork() as uow:
246                repo = IrisAnalysisRepository(uow)
247                fresh = repo.get_by_id(analysis_id)
248                if fresh:
249                    fresh.status = "cancelled"
250                    fresh.finished_at = datetime.now()
251                    repo.update(fresh)
252            self.logger.info(f"Analysis {analysis_id} cancelled by user {user_id}")
253        return cancelled

377    def _run_analysis(self, analysis_id: int, raw_headers: str) -> None:
378        """Background task: execute all rules and persist results.
379
380        This is the function submitted to SeQueue.  It:
381        1. Marks the analysis as ``running``.
382        2. Parses the raw header text.
383        3. Iterates over every registered rule, collects RuleResults.
384        4. Updates the SeQueue task progress after each rule.
385        5. Computes the total score and verdict.
386        6. Persists the final state (``finished`` + score + verdict).
387
388        If cancellation is detected between rule executions, the task
389        exits early without saving results.
390        """
391        self.logger.info(f"Starting analysis {analysis_id}")
392
393        try:
394            with UnitOfWork() as uow:
395                repo = IrisAnalysisRepository(uow)
396                fresh = repo.get_by_id(analysis_id)
397                if fresh:
398                    fresh.status = "running" # type: ignore
399                    fresh.started_at = datetime.now() # type: ignore
400                    repo.update(fresh)
401        except Exception as e:
402            self.logger.error(f"Failed to mark analysis {analysis_id} as running: {e}")
403            self._fail_analysis(analysis_id)
404            return
405
406        headers = parse_raw_headers(raw_headers)
407        self._validate_headers_parsed(headers)
408
409        rules_defs = iris_rules.get_rules()
410        total_rules = len(rules_defs)
411        results: List[RuleResult] = []
412        sequeue = SeQueue.get_instance()
413
414        for idx, rule_def in enumerate(rules_defs):
415            if self._is_cancelled(analysis_id):
416                self.logger.info(f"Analysis {analysis_id} was cancelled")
417                return
418
419            try:
420                result = rule_def["func"](headers)
421            except Exception as e:
422                self.logger.error(f"Rule '{rule_def['name']}' failed for analysis {analysis_id}: {e}")
423                result = RuleResult(
424                    score=0, verdict="error",
425                    details={"error": str(e)},
426                    recommendation=f"La regla '{rule_def['name']}' falló durante la ejecución.",
427                )
428
429            self._persist_rule_result(analysis_id, rule_def, result, idx)
430            results.append(result)
431
432            progress = int(((idx + 1) / total_rules) * 100)
433            sq_task = sequeue.get_task_by_external_id(
434                f"iris-analysis:{analysis_id}", category="iris.analyze"
435            )
436            if sq_task:
437                sequeue.update_progress(sq_task.id, progress)
438
439        total_score = sum(r.score for r in results)
440        verdict = self._determine_verdict(total_score)
441
442        try:
443            with UnitOfWork() as uow:
444                repo = IrisAnalysisRepository(uow)
445                fresh = repo.get_by_id(analysis_id)
446                if fresh:
447                    fresh.status = "finished" # type: ignore
448                    fresh.total_score = total_score # type: ignore
449                    fresh.verdict = verdict # type: ignore
450                    fresh.finished_at = datetime.now() # type: ignore
451                    repo.update(fresh)
452        except Exception as e:
453            self.logger.error(f"Failed to finalise analysis {analysis_id}: {e}")
454            self._fail_analysis(analysis_id)
455            return
456
457        self.logger.info(f"Analysis {analysis_id} completed: score={total_score}, verdict={verdict}")
458

211
